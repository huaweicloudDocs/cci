**请求示例：**

```
{
    "apiVersion": "apps/v1",
    "kind": "StatefulSet",
    "metadata": {
        "name": "statefulset-test"
    },
    "spec": {
        "replicas": 3,
        "selector": {
            "matchLabels": {
                "app": "statefulset-test"
            }
        },
        "template": {
            "metadata": {
                "labels": {
                    "app": "statefulset-test"
                }
            },
            "spec": {
                "containers": [
                    {
                        "image": "redis",
                        "name": "container-0",

                            "limits": {
                                "cpu": "500m",

                            },
                            "requests": {
                                "cpu": "500m",
                                "memory": "1024Mi"
                            }
                        }
                    }
                ],
                "imagePullSecrets": [
                    {
                        "name": "imagepull-secret"
                    }
                ],
                "priority": 0

    }
}
```
